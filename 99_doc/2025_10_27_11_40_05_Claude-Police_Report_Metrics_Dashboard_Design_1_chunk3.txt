// ============================================
// OFFICER NAME STANDARDIZATION
// ============================================

let
    Source = Excel.CurrentWorkbook(){[Name="INFOCOP_OFFICER_CHARGE_TYPE"]}[Content],
    
    // Standardize officer names for better matching
    StandardizedNames = Table.AddColumn(Source, "Officer Name Standardized", each
        let
            CleanName = Text.Trim(Text.Clean([Officer Name])),
            ProperCase = Text.Proper(CleanName),
            // Handle common name variations
            RemoveMiddleInitials = Text.Replace(Text.Replace(ProperCase, " Jr.", ""), " Sr.", ""),
            RemoveExtraSpaces = Text.Replace(RemoveExtraSpaces, "  ", " ")
        in
            RemoveExtraSpaces
    , type text),
    
    // Create lookup table for manual name corrections
    NameMappingTable = Table.FromRows({
        {"Johnson, Michael", "Johnson, Mike"},
        {"Williams, Robert", "Williams, Rob"},
        {"Smith, Jonathan", "Smith, John"}
        // Add more mappings as needed
    }, {"InfoCop_Name", "Assignment_Name"}),
    
    // Apply manual corrections
    ApplyMappings = Table.AddColumn(StandardizedNames, "Final_Officer_Name", each
        let
            LookupResult = Table.SelectRows(NameMappingTable, each [InfoCop_Name] = [Officer Name Standardized])
        in
            if Table.RowCount(LookupResult) > 0 
            then LookupResult{0}[Assignment_Name]
            else [Officer Name Standardized]
    , type text)
in
    ApplyMappings
