import subprocess
from pathlib import Path
from datetime import datetime

__version__ = "1.0.0"

# Folder structure
INPUT_DIR = Path("01_input")
OUTPUT_DIR = Path("02_output")
KB_DIR = Path("03_knowledge_base")

def ensure_folders():
    """Create folders if they don't exist"""
    INPUT_DIR.mkdir(exist_ok=True)
    OUTPUT_DIR.mkdir(exist_ok=True)
    KB_DIR.mkdir(exist_ok=True)

def get_input_files():
    """Get all ZIP and JSON files from input folder"""
    files = []
    if INPUT_DIR.exists():
        for ext in ['*.zip', '*.json']:
            files.extend(INPUT_DIR.glob(ext))
    return sorted(files)

def generate_output_name(input_file: Path) -> Path:
    """Generate output filename with timestamp"""
    timestamp = datetime.now().strftime("%Y%m%d-%H%M%S")
    base = input_file.stem
    # Remove any existing -FIXED suffix
    if base.endswith('-FIXED'):
        base = base[:-6]
    return OUTPUT_DIR / f"{base}-FIXED-{timestamp}.zip"

def process_file(input_file: Path, output_file: Path) -> bool:
    """Process a single file with patch_conversations.py"""
    print(f"\n{'='*70}")
    print(f"ğŸ“¥ Processing: {input_file.name}")
    print(f"ğŸ“¤ Output: {output_file.name}")
    print(f"{'='*70}\n")
    
    cmd = [
        sys.executable,
        "patch_conversations.py",
        str(input_file),
        str(output_file)
    ]
    
    try:
        result = subprocess.run(cmd, check=True, capture_output=False)
        print(f"\nâœ… Successfully processed: {input_file.name}")
        return True
    except subprocess.CalledProcessError as e:
        print(f"\nâŒ Error processing {input_file.name}: {e}")
        return False

def build_knowledge_base(output_file: Path, incremental: bool = False) -> bool:
    """Build knowledge base from processed file"""
    print(f"\n{'='*70}")
    print(f"ğŸ—„ï¸  Building knowledge base from: {output_file.name}")
    print(f"{'='*70}\n")
    
    cmd = [
        sys.executable,
        "claude_knowledge_base.py",
        str(output_file)
    ]
    
    if incremental and (KB_DIR / "claude_knowledge_base.db").exists():
        cmd.append("--incremental")
        print("ğŸš€ Using incremental mode (87-90% faster!)") try:
        result = subprocess.run(cmd, check=True, capture_output=False)
        print(f"\nâœ… Knowledge base built successfully")
        return True
    except subprocess.CalledProcessError as e:
        print(f"\nâŒ Error building knowledge base: {e}")
        return False

def main():
    parser = argparse.ArgumentParser(
        description="Process all Claude exports in 01_input/ folder"
    )
    parser.add_argument(
        "--build-kb",
        action="store_true",
        help="Also build/update knowledge base after processing"
    )
    parser.add_argument(
        "--incremental",
        action="store_true",
        help="Use incremental mode for knowledge base (faster)"
    )
    parser.add_argument(
        "--version",
        action="version",
        version=f"%(prog)s {__version__}"
    )
    
    args = parser.parse_args()
    
    # Ensure folder structure exists
    ensure_folders()
    
    # Get input files
    input_files = get_input_files()
    
    if not input_files:
        print(f"\nâš ï¸  No files found in {INPUT_DIR}/")
        print(f"\nğŸ’¡ Drop your Claude export ZIP or JSON files in the {INPUT_DIR}/ folder")
        print("   Then run this script again.\n")
        return 1
    
    print(f"\nğŸ¯ Found {len(input_files)} file(s) to process:")
    for f in input_files:
        print(f"   â€¢ {f.name}")
    print()
    
    # Process each file
    processed_files = []
    for input_file in input_files:
        output_file = generate_output_name(input_file)
        if process_file(input_file, output_file):
            processed_files.append(output_file)
    
    # Summary
    print(f"\n{'='*70}")
    print(f"ğŸ“Š Processing Summary")
    print(f"{'='*70}")
    print(f"âœ… Processed: {len(processed_files)}/{len(input_files)} files")
    print(f"ğŸ“‚ Output folder: {OUTPUT_DIR}/")
    
    if processed_files:
        print("\nğŸ“¤ Fixed files:")
        for f in processed_files:
            print(f"   â€¢ {f.name}")
    
    # Build knowledge base if requested
    if args.build_kb and processed_files:
        # Use the most recent processed file
        latest = processed_files[-1]
        print(f"\nğŸ—„ï¸  Building knowledge base...")
        build_knowledge_base(latest, incremental=args.incremental)
    
    print(f"\n{'='*70}")
    print("âœ¨ Done!") print(f"{'='*70}\n")
    
    return 0

if __name__ == "__main__":
    sys.exit(main())
