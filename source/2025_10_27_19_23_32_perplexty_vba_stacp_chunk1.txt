Private Sub Worksheet_Change(ByVal Target As Range)
    Dim OldValue As String
    Dim NewValue As String
    Dim Separator As String
    Separator = ", " ' Define the separator used for multiple selections

    On Error Resume Next

    ' Check if the changed cell is within the validated dropdown range
    If Not Intersect(Target, Me.Range("AI:AI5")) Is Nothing Then
        If Target.Validation.Type = xlValidateList Then
            Application.EnableEvents = False
            NewValue = Target.Value
            Application.Undo
            OldValue = Target.Value
            ' If the new value is empty, allow clearing the cell
            If NewValue = "" Then
                Target.Value = ""
            Else
                ' If there's an existing value, append the new selection
                If OldValue = "" Then
                    Target.Value = NewValue
                Else
                    Target.Value = OldValue & Separator & NewValue
                End If
            End If
            Application.EnableEvents = True
        End If
    End If

    On Error GoTo 0
End Sub
